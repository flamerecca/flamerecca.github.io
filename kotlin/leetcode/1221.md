## Kotlin 參考解答

```kotlin
class Solution {
    fun balancedStringSplit(s: String): Int {
        var split = 0
        var rc = 0
        var lc = 0
        for (c in s) {
            if (c == 'R') rc++
            if (c == 'L') lc++
            if (lc == rc) {
                rc = 0
                lc = 0
                split++
            }
        }
        return split
    }
}
```

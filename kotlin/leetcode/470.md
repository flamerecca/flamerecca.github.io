```kotlin
class Solution : SolBase() {
    fun rand10(): Int {
        while(true) {
            val a = rand7()
            val b = rand7()
            val idx = 7 * (a - 1) + b
            if (idx <= 40) {
                return 1 + (idx - 1) % 10;
            }
        }
    }
}
```
